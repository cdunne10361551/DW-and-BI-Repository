DROP TABLE FACT_SALES;

CREATE TABLE DBO.FACT_SALES(FACTID INT IDENTITY (1,1) PRIMARY KEY,
							ORDERID INT,
							PRODUCTID INT,
							EMPLOYEEID INT,
							CUSTOMERID NCHAR(5),
							ORDERDATE DATETIME,
							UNITPRICE INT,
							QUANTITY INT,
							DISCOUNT INT);

INSERT INTO DBO.FACT_SALES(ORDERID, 
							PRODUCTID,
							EMPLOYEEID,
							CUSTOMERID,
							ORDERDATE,
							UNITPRICE,
							QUANTITY,
							DISCOUNT)
					SELECT d_ord.ORDERID,
							d_ord.PRODUCTID,
							o.EMPLOYEEID,
							o.CUSTOMERID,
							o.ORDERDATE,
							d_ord.UNITPRICE,
							d_ord.QUANTITY,
							d_ord.DISCOUNT
					FROM NORTHWND.DBO.[Order Details] d_ord,
						NORTHWND.DBO.Orders o
					WHERE d_ord.ORDERID = o.ORDERID;


SELECT * FROM FACT_SALES;
